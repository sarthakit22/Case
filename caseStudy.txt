package com.atcs.casestudy;
import java.util.Scanner;

public class CountLettersInArray {
	Scanner sc=new Scanner(System.in);
  
    
	
     char[] createArrays() {      //input the lower case 
    		System.out.println("Enter character ");
    	 	char array[]=sc.next().toCharArray();
    	return array;
	}
    
    void display(char a[]) {
    	for(int i=0;i<a.length;i++) {
    	System.out.println(a[i]);
    	}
    }
    
   int[] randomCharacter(char a[]) {
	   int l=a.length;
	   int count[];
	   count=new int[l];
	   for(int i=0;i<l;i++) {
		   count[i]=1;
		   for(int j=i+1;j<l;j++) {
		   if(a[i]==a[j]) {
			   count[i]++;
			   a[j]='0';
		   }
	       }
       }
	   return count;
   }
   
   void displayCount(char a[],int count[]) {
	   System.out.println("----Character and repeating counts----");
	   int ii=a.length;
	   for(int i=0;i<ii;i++) {
		   if(a[i] !=' '  &&  a[i] !='0') {
	       System.out.println("The Number of times "+a[i]+" has occured= "+count[i]);
           }
       }
   }
  
}
class ArrayMain{
	public static void main(String args[]) {
	Scanner sc=new Scanner(System.in);
	CountLettersInArray obj=new CountLettersInArray();
	
	char arry[]=obj.createArrays();
	obj.display(arry);
	
	//Pick the random character
	int c[]=obj.randomCharacter(arry);         //now we get the character we need to count
	
	obj.displayCount(arry,c);
	}
}
